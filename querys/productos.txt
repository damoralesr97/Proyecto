CREATE TABLE `productos` ( 
  `prod_id` int(11) NOT NULL AUTO_INCREMENT, 
  `prod_imagen` varchar(255) NOT NULL,
  `prod_nombre` varchar(100) NOT NULL,
  `prod_descripcion` varchar(255) NOT NULL,
  `prod_precio` varchar(11) NOT NULL, 
  `prod_stock` varchar(11) NOT NULL,  
  `prod_categoria` varchar(255) NOT NULL, 
  `prod_eliminado` varchar(1) NOT NULL DEFAULT 'N',
  `prod_rating` varchar(255) NOT NULL, 
   PRIMARY KEY (`prod_id`)
) ENGINE=InnoDB  DEFAULT CHARSET=utf8 AUTO_INCREMENT=1 ;
